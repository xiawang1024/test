function startMove(obj,attr,iTarget,fn){ clearInterval(obj.timer); obj.timer=setInterval(function(){ var current; if(attr=="opacity"){ current=parseFloat(getStyle(obj,attr))*100; }else{ current=parseInt(getStyle(obj,attr)); } var speed=(iTarget-current)/8; speed=speed>0?Math.ceil(speed):Math.floor(speed); if(iTarget===current){ clearInterval(obj.timer); if(fn){ fn(); } }else{ if(attr==="opacity"){ obj.style[attr]=(current+speed)/100; }else{ obj.style[attr]=current+speed+"px"; } } },25); } function getStyle(obj,attr){ if(obj.currentStyle){ return obj.currentStyle[attr]; }else{ return window.getComputedStyle(obj,null)[attr]; } }
